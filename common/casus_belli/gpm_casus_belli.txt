# ROOT = you
# FROM = enemy

cb_gpm_claim_relic = {
	show_in_diplomacy = yes
	show_notification = yes
	potential = {
		is_neighbor_of = from
		OR = {
			gpm_is_country_type_default = yes
			gpm_is_country_type_awakened_fallen_empire = yes
			# is_berserk_fallen_machine_empire = yes
		}
		NOT = { is_in_federation_with = from }
		has_intel_level = {
			who = FROM
			category = government
			level >= 2
		}
		from = {
			# a faster check before asking for each supported relic
			if = {
				limit = {
					has_global_flag = gigastructures_enabled 
					has_relic = r_giga_ai_savings
				}
				num_owned_relics > 1
			}
			else = {
				num_owned_relics > 0
			}
			gpm_has_wargoal_relic = yes
		}
	}
	is_valid = {
		# has_total_war_cb = no
		NOT = { is_overlord_to = FROM }
		OR = {
			is_subject = no
			overlord = { NOT = { is_same_value = from } }
		}
	}
}
